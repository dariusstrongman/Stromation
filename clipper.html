Video playback not configured. Set CLIPS_BASE_URL inside clipper.html (line 1) to a public HTTPS URL that serves /mnt/clipper-storage/clipper/clips/ on the EC2 box. Until then you'll see paths instead of embedded videos.
Quick nginx setup (1-time) 
sudo apt install nginx apache2-utils -y

# basic auth
sudo htpasswd -c /etc/nginx/.htpasswd_clips darius

# /etc/nginx/sites-available/clipper-files
server {
  listen 443 ssl;
  server_name clips.stromation.com;
  # ssl_certificate ... (use certbot)
  auth_basic "Clips";
  auth_basic_user_file /etc/nginx/.htpasswd_clips;
  root /mnt/clipper-storage/clipper;
  autoindex off;
  location / { try_files $uri =404; }
}

# point DNS A record clips.stromation.com to the EC2 IP
# then: sudo certbot --nginx -d clips.stromation.com
# then set CLIPS_BASE_URL = 'https://clips.stromation.com/' in this file
